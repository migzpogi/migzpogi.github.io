linkRunning a local Ansible playbook On Ubuntu 20.04

linkInstall Ansible

1link$sudo apt-add-repository ppa:ansible/ansible

2link$sudo apt update

3link$sudo apt install ansible

linkCreate an Ansible playbook

local.yml
1link---

2link- name: "Playing with Ansible and Git"

3link  hosts: localhost

4link  connection: local 

5link  tasks:

6link

7link  - name: "just execute a ls -lrt command"

8link    shell: "ls -lrt"

9link    register: "output"

10link

11link  - debug: var=output.stdout_lines

linkRun the playbook

1link$ansible-playbook local.yml

linkRunning the playbook with an inventory file

1link$ansible-playbook -i inventory_file_name local.yml

linkSample inventory file

1link[dev]

2linkdjango_ap_south ansible_host=#{server-django-ap-south}# ansible_user=#{server-django-ap-south-user}# ansible_ssh_pass=#{server-django-ap-south-pass}#
